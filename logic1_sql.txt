create table ASSOCIAZIONE (
	Nome char(30) not null,
	Città char(25) not null,
	Via char(25) not null,
	Numero_civico numeric(6) not null,
	primary key (Nome));

create table SPEDIZIONE (
	Codice varchar(10) not null,
	Data date not null,
	primary key (Codice));

create table AVVISTAMENTO (
	Codice varchar(10) not null,
	Numero numeric(4) not null,
    Codice_spedizione varchar(10) not null,
	primary key (Codice),
    foreign key (Codice_spedizione) references SPEDIZIONE(Codice),
    unique (Numero));

create table CASA_PRODUTTRICE (
	Nome char(30) not null,
	Città char(25) not null,
	Via char(25) not null,
	Numero_civico numeric(6) not null,
	primary key (Nome));

create table GRUPPO_DI_ESPLORAZIONE (
	ID varchar(5) not null,
	Nome char(20) not null,
    Nome_associazione char(30) not null,
    foreign key (Nome_associazione) references ASSOCIAZIONE(Nome),
	unique (ID));

create table ISTITUTO_PROFESSIONALE_SPECIALIZZATO (
	Nome char(30) not null,
	ID varchar(10) not null,
	Città char(25) not null,
	Via char(25) not null,
	Numero_civico numeric(6) not null,
	primary key (ID));

create table LABORATORIO (
	Nome char(30) not null,
	ID varchar(10) not null,
	Città char(25) not null,
	Via char(25) not null,
	Numero_civico numeric(6) not null,
	primary key (ID));

create table LICENZA_MANUTENTORE (
	ID varchar(10) not null,
	Data_rilascio date not null,
	Data_rinnovo date,
	Data_scadenza date not null,
	primary key (ID));

create table LICENZA_OPERATORE_ROV (
	ID varchar(10) not null,
	Data_rilascio date not null,
	Data_rinnovo date,
	Data_scadenza date not null,
	primary key (ID));

create table MARE (
	Nome char(20) not null,
	primary key (Nome));

create table MATERIALE (
	Nome char(20) not null,
	Quantità numeric(3) not null,
	Descrizione char(200) not null);

create table ORGANISMO (
	Specie char(30) not null,
	Dimensioni numeric(2) not null,
	Profondità numeric(5) not null,
	Descrizione_comportamento text(500) not null);
    
create table PERSONA (
	Nome char(1) not null,
	Cognome char(1) not null,
	Codice_fiscale char(1) not null,
	Indirizzo char(1) not null,
    ID_gruppo varchar(5),
    Nome_casa_produttrice char(30),
	ID_manutentore varchar(5),
	ID_operatore varchar(5),
	primary key (Codice_fiscale),
    foreign key (ID_gruppo) references GRUPPO_DI_ESPLORAZIONE(ID),
    foreign key (Nome_casa_produttrice) references CASA_PRODUTTRICE(Nome));

create table PRELIEVO (
	Numero numeric(3) not null,
	Codice varchar(10) not null,
    Codice_spedizione varchar(10) not null,
	primary key (Codice),
    foreign key (Codice_spedizione) references SPEDIZIONE(Codice));

create table ROV (
	Targa varchar(7) not null,
	Numero_serie varchar(10) not null,
    Casa_produttrice char(30),
	primary key (Targa),
    foreign key (Casa_produttrice) references CASA_PRODUTTRICE(Nome));
